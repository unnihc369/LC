## 1) First non-repeating character

### Approach A: Frequency map
```tsx
export function firstNonRepeatingCharMap(s: string): string | null {
  const freq = new Map<string, number>();
  for (const ch of s) freq.set(ch, (freq.get(ch) ?? 0) + 1);
  for (const ch of s) if (freq.get(ch) === 1) return ch;
  return null;
}
```

### Approach B: `indexOf` + `lastIndexOf`
```tsx
export function firstNonRepeatingCharIndex(s: string): string | null {
  for (const ch of s) if (s.indexOf(ch) === s.lastIndexOf(ch)) return ch;
  return null;
}
```

## 2) Rotate array right by k (no extra array)

### Approach A: Reverse in-place (optimal)
```tsx
export function rotateRightReverse(nums: number[], k: number): number[] {
  const n = nums.length;
  if (n === 0) return nums;
  const step = ((k % n) + n) % n;
  reverse(nums, 0, n - 1);
  reverse(nums, 0, step - 1);
  reverse(nums, step, n - 1);
  return nums;
}

function reverse(arr: number[], l: number, r: number): void {
  while (l < r) {
    const t = arr[l];
    arr[l++] = arr[r];
    arr[r--] = t;
  }
}
```

### Approach B: Repeated 1-step rotation (in-place, slower)
```tsx
export function rotateRightRepeated(nums: number[], k: number): number[] {
  const n = nums.length;
  if (n === 0) return nums;
  const step = ((k % n) + n) % n;
  for (let i = 0; i < step; i++) {
    const last = nums[n - 1];
    for (let j = n - 1; j > 0; j--) nums[j] = nums[j - 1];
    nums[0] = last;
  }
  return nums;
}
```

## 3) Longest substring without repeating characters

### Approach A: Sliding window + Set
```tsx
export function longestSubstringNoRepeatSet(s: string): number {
  const seen = new Set<string>();
  let left = 0;
  let best = 0;
  for (let right = 0; right < s.length; right++) {
    while (seen.has(s[right])) seen.delete(s[left++]);
    seen.add(s[right]);
    best = Math.max(best, right - left + 1);
  }
  return best;
}
```

### Approach B: Sliding window + last index map
```tsx
export function longestSubstringNoRepeatMap(s: string): number {
  const last = new Map<string, number>();
  let left = 0;
  let best = 0;
  for (let right = 0; right < s.length; right++) {
    const ch = s[right];
    const prev = last.get(ch);
    if (prev !== undefined && prev >= left) left = prev + 1;
    last.set(ch, right);
    best = Math.max(best, right - left + 1);
  }
  return best;
}
```

## 4) Move all zeroes to end (stable order)

### Approach A: Swap with write pointer
```tsx
export function moveZeroesSwap(nums: number[]): number[] {
  let write = 0;
  for (let read = 0; read < nums.length; read++) {
    if (nums[read] !== 0) {
      const t = nums[write];
      nums[write] = nums[read];
      nums[read] = t;
      write++;
    }
  }
  return nums;
}
```

### Approach B: Compact then fill zeros
```tsx
export function moveZeroesFill(nums: number[]): number[] {
  let write = 0;
  for (let i = 0; i < nums.length; i++) if (nums[i] !== 0) nums[write++] = nums[i];
  while (write < nums.length) nums[write++] = 0;
  return nums;
}
```

## 5) Check anagram using frequency

### Approach A: Hash map frequency
```tsx
export function isAnagramMap(s: string, t: string): boolean {
  if (s.length !== t.length) return false;
  const freq = new Map<string, number>();
  for (const ch of s) freq.set(ch, (freq.get(ch) ?? 0) + 1);
  for (const ch of t) {
    const cur = freq.get(ch);
    if (cur === undefined) return false;
    if (cur === 1) freq.delete(ch);
    else freq.set(ch, cur - 1);
  }
  return freq.size === 0;
}
```

### Approach B: 26-length array (lowercase only)
```tsx
export function isAnagramArrayLowercase(s: string, t: string): boolean {
  if (s.length !== t.length) return false;
  const count = new Array<number>(26).fill(0);
  for (let i = 0; i < s.length; i++) {
    count[s.charCodeAt(i) - 97]++;
    count[t.charCodeAt(i) - 97]--;
  }
  for (const x of count) if (x !== 0) return false;
  return true;
}
```

## 6) Index of second highest number

### Approach A: Distinct second highest
```tsx
export function indexOfSecondHighestDistinct(nums: number[]): number {
  if (nums.length < 2) return -1;
  let maxVal = -Infinity, maxIdx = -1;
  let secondVal = -Infinity, secondIdx = -1;
  for (let i = 0; i < nums.length; i++) {
    const x = nums[i];
    if (x > maxVal) {
      secondVal = maxVal; secondIdx = maxIdx;
      maxVal = x; maxIdx = i;
    } else if (x < maxVal && x > secondVal) {
      secondVal = x; secondIdx = i;
    }
  }
  return secondIdx;
}
```

### Approach B: Allow tie on max (second occurrence)
```tsx
export function indexOfSecondHighestAllowTie(nums: number[]): number {
  if (nums.length < 2) return -1;
  let maxVal = -Infinity;
  for (const x of nums) if (x > maxVal) maxVal = x;
  let seen = 0;
  for (let i = 0; i < nums.length; i++) {
    if (nums[i] === maxVal) {
      seen++;
      if (seen === 2) return i;
    }
  }
  return indexOfSecondHighestDistinct(nums);
}
```

## 8) Maximum sum of any contiguous subarray

```tsx
export function maxSubarraySum(nums: number[]): number {
  let best = -Infinity;
  let cur = 0;
  for (const x of nums) {
    cur = Math.max(x, cur + x);
```

## 9) Reverse word order without using split()

```tsx
export function reverseWordOrderNoSplit(s: string): string {
  const words: string[] = [];
  let i = 0;
  while (i < s.length) {
    while (i < s.length && s[i] === " ") i++;
    if (i >= s.length) break;
    const start = i;
    while (i < s.length && s[i] !== " ") i++;
    words.push(s.slice(start, i));
  }
  let out = "";
  for (let j = words.length - 1; j >= 0; j--) {
    out += words[j];
    if (j > 0) out += " ";
  }
  return out;
}
```

## 10) All unique pairs that sum to target

```tsx
export function uniquePairsWithTarget(nums: number[], target: number): number[][] {
  const seen = new Set<number>();
  const pairs = new Set<string>();
  const out: number[][] = [];
  for (const x of nums) {
    const y = target - x;
    if (seen.has(y)) {
      const a = Math.min(x, y);
      const b = Math.max(x, y);
      const key = `${a},${b}`;
      if (!pairs.has(key)) {
        pairs.add(key);
        out.push([a, b]);
      }
    }
    seen.add(x);
  }
  return out;
}
```

## 11) Sentence palindrome ignoring spaces/punctuation

```tsx
export function isSentencePalindrome(s: string): boolean {
  let l = 0, r = s.length - 1;
  while (l < r) {
    while (l < r && !isAlphaNum(s[l])) l++;
    while (l < r && !isAlphaNum(s[r])) r--;
    if (s[l].toLowerCase() !== s[r].toLowerCase()) return false;
    l++;
    r--;
  }
  return true;
}

function isAlphaNum(ch: string): boolean {
  const code = ch.charCodeAt(0);
  return (
    (code >= 48 && code <= 57) ||
    (code >= 65 && code <= 90) ||
    (code >= 97 && code <= 122)
  );
}
```

### Approach A: Boyer-Moore voting
```tsx
export function majorityElement(nums: number[]): number | null {
  if (nums.length === 0) return null;
  let candidate = 0;
  let count = 0;
  for (const x of nums) {
    if (count === 0) candidate = x;
    count += x === candidate ? 1 : -1;
  }
  return candidate;
}
```

## 17) Longest palindromic substring

```tsx
export function longestPalindromicSubstring(s: string): string {
  if (s.length < 2) return s;
  let bestL = 0, bestR = 0;
  for (let i = 0; i < s.length; i++) {
    expand(i, i);
    expand(i, i + 1);
  }
  return s.slice(bestL, bestR + 1);

  function expand(l: number, r: number): void {
    while (l >= 0 && r < s.length && s[l] === s[r]) {
      if (r - l > bestR - bestL) {
        bestL = l;
        bestR = r;
      }
      l--;
      r++;
    }
  }
}
```

## 20) Merge two sorted arrays

```tsx
export function mergeSortedArrays(a: number[], b: number[]): number[] {
  const out: number[] = [];
  let i = 0, j = 0;
  while (i < a.length && j < b.length) {
    if (a[i] <= b[j]) out.push(a[i++]);
    else out.push(b[j++]);
  }
  while (i < a.length) out.push(a[i++]);
  while (j < b.length) out.push(b[j++]);
  return out;
}
```

## 23) All permutations of a string

```tsx
export function permutations(str: string): string[] {
  const chars = str.split("");
  const out: string[] = [];
  backtrack(0);
  return out;

  function backtrack(i: number): void {
    if (i === chars.length) {
      out.push(chars.join(""));
      return;
    }
    const used = new Set<string>();
    for (let j = i; j < chars.length; j++) {
      if (used.has(chars[j])) continue;
      used.add(chars[j]);
      [chars[i], chars[j]] = [chars[j], chars[i]];
      backtrack(i + 1);
      [chars[i], chars[j]] = [chars[j], chars[i]];
    }
  }
}
```

## 24) Product except self

```tsx
export function productExceptSelf(nums: number[]): number[] {
  const n = nums.length;
  const out = new Array<number>(n).fill(1);
  for (let i = 1; i < n; i++) out[i] = out[i - 1] * nums[i - 1];
  let suffix = 1;
  for (let i = n - 1; i >= 0; i--) {
    out[i] *= suffix;
    suffix *= nums[i];
  }
  return out;
}
```

## 25) Check if two strings follow same pattern

```tsx
export function samePattern(a: string, b: string): boolean {
  if (a.length !== b.length) return false;
  const ab = new Map<string, string>();
  const ba = new Map<string, string>();
  for (let i = 0; i < a.length; i++) {
    const x = a[i], y = b[i];
    if ((ab.has(x) && ab.get(x) !== y) || (ba.has(y) && ba.get(y) !== x)) return false;
    ab.set(x, y);
    ba.set(y, x);
  }
  return true;
}
```

```tsx
export function partitionEvenOdd(nums: number[]): number[] {
  let l = 0, r = nums.length - 1;
  while (l < r) {
    while (l < r && nums[l] % 2 === 0) l++;
    while (l < r && nums[r] % 2 !== 0) r--;
    if (l < r) [nums[l], nums[r]] = [nums[r], nums[l]];
  }

}
```

## 27) Validate balanced brackets

```tsx
export function isBalancedBrackets(s: string): boolean {
  const st: string[] = [];
  const closeToOpen = new Map<string, string>([
    [")", "("], ["]", "["], ["}", "{"],
  ]);
  for (const ch of s) {
    if (closeToOpen.has(ch)) {
      if (st.pop() !== closeToOpen.get(ch)) return false;
    } else if (ch === "(" || ch === "[" || ch === "{") {
      st.push(ch);
    }
  }
  return st.length === 0;
}
```

## 28) Return all subsets of an array

```tsx
export function subsets(nums: number[]): number[][] {
  const out: number[][] = [];
  const cur: number[] = [];
  backtrack(0);
  return out;

  function backtrack(i: number): void {
    if (i === nums.length) {
      out.push([...cur]);
      return;
    }
    backtrack(i + 1);
    cur.push(nums[i]);
    backtrack(i + 1);
    cur.pop();
  }
}
```

```tsx
        if (p[j - 2] === "." || p[j - 2] === s[i - 1]) {
          dp[i][j] = dp[i][j] || dp[i - 1][j];
        }
      } else if (p[j - 1] === "." || p[j - 1] === s[i - 1]) {
        dp[i][j] = dp[i - 1][j - 1];
      }
    }
  }
  return dp[m][n];
}
```

```tsx
export function intersection(a: number[], b: number[]): number[] {
  const sb = new Set(b);
  const seen = new Set<number>();
  const out: number[] = [];

    if (sb.has(x) && !seen.has(x)) {

    }
  }
  return out;
```

## 90) Pair with largest product

```tsx
export function largestProductPair(nums: number[]): [number, number] | null {
  if (nums.length < 2) return null;
  let max1 = -Infinity, max2 = -Infinity, min1 = Infinity, min2 = Infinity;
  for (const x of nums) {
    if (x >= max1) {
      max2 = max1;
      max1 = x;
    } else if (x > max2) max2 = x;
    if (x <= min1) {
      min2 = min1;
      min1 = x;
    } else if (x < min2) min2 = x;
  }
  return min1 * min2 > max1 * max2 ? [min1, min2] : [max1, max2];
}
```

## 91) Reverse only alphabets (mirror)

```tsx
export function reverseOnlyAlphabets(s: string): string {
  const arr = s.split("");
  const isLetter = (ch: string) => /[a-zA-Z]/.test(ch);
  let l = 0, r = arr.length - 1;
  while (l < r) {
    while (l < r && !isLetter(arr[l])) l++;
    while (l < r && !isLetter(arr[r])) r--;
    if (l < r) [arr[l], arr[r]] = [arr[r], arr[l]];
    l++;
    r--;
  }
  return arr.join("");
}
```

## 94) Median of two sorted arrays

```tsx
export function findMedianSortedArrays(a: number[], b: number[]): number {
  const m = a.length, n = b.length;
  const need = Math.floor((m + n + 1) / 2);
  let i = 0, j = 0;
  let last = 0;
  for (let k = 0; k < need; k++) {
    if (i < m && (j >= n || a[i] <= b[j])) last = a[i++];
    else last = b[j++];
  }
  if ((m + n) % 2 === 1) return last;
  let next: number;
  if (i < m && (j >= n || a[i] <= b[j])) next = a[i];
  else next = b[j];
  return (last + next) / 2;
}
```

## 96) Sort by frequency (high first)

```tsx
export function sortByFrequency(nums: number[]): number[] {
  const freq = new Map<number, number>();
  for (const x of nums) freq.set(x, (freq.get(x) ?? 0) + 1);
  return [...nums].sort((a, b) => {
    const fa = freq.get(a)!, fb = freq.get(b)!;
    if (fb !== fa) return fb - fa;
    return a - b;
  });
}
```

## 98) Longest mountain subarray

```tsx
export function longestMountain(nums: number[]): number {
  let best = 0;
  for (let i = 1; i < nums.length - 1; i++) {
    if (nums[i] > nums[i - 1] && nums[i] > nums[i + 1]) {
      let l = i, r = i;
      while (l > 0 && nums[l - 1] < nums[l]) l--;
      while (r < nums.length - 1 && nums[r] > nums[r + 1]) r++;
      best = Math.max(best, r - l + 1);
    }
  }
  return best;
}
```
